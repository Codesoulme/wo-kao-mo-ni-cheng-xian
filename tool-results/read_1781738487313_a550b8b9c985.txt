   115→   115→function buildAdvancePrompt(ctx: EngineStateContext, isFateNode: boolean): string {
   116→   116→  const sc = ctx.character;
   117→   117→  const elements = `金${sc.elements.metal}/木${sc.elements.wood}/水${sc.elements.water}/火${sc.elements.fire}/土${sc.elements.earth}`;
   118→   118→  const statusList = ctx.activeStatuses.length
   119→   119→    ? ctx.activeStatuses.map(s => `- ${s.name}（${s.category}，${s.rarity}）：${s.description}`).join('\n')
   120→   120→    : '无';
   121→   121→  const invList = ctx.inventory.length
   122→   122→    ? ctx.inventory.map(i => `- [id:${i.id}] ${i.name}（${i.rarity}/${i.item_type}）：${i.description}${i.effects?.length ? '；效果：' + i.effects.map(e => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   123→   123→    : '无';
   124→   124→  const eqList = ctx.equipped && Object.keys(ctx.equipped).length
   125→   125→    ? Object.entries(ctx.equipped).map(([slot, it]: [string, any]) => `- [槽位:${slot}][id:${it.id}] ${it.name}（${it.rarity}/${it.item_type}）：${it.description}${it.effects?.length ? '；效果：' + it.effects.map((e: any) => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   126→   126→    : '无';
   127→   127→  const recentEvts = ctx.recentEvents.length
   128→   128→    ? ctx.recentEvents.map(e => `${e.age}岁：${e.title}——${e.narrative.slice(0, 80)}`).join('\n')
   129→   129→    : '无';

   131→   131→    ? ctx.longTermMemory.map(m => `- ${m}`).join('\n')

   138→   138→寿元：${sc.lifespan}岁（剩余约${sc.lifespan - sc.age}岁）
   139→   139→灵根：${sc.rootDetail || sc.spiritualRoot}

   141→   141→修为：${sc.cultivationExp}/${sc.expToBreak}（修炼速度：${multDesc}，你给出的 cultivationExp 正向增量会被该倍率放大）
   142→   142→五行倾向：${elements}
   143→   143→生命：${sc.hp}/${sc.maxHp}  灵力：${sc.mp}/${sc.maxMp}
   144→   144→攻击：${sc.attack}  防御：${sc.defense}  速度：${sc.speed}
   145→   145→气运：${sc.luck}  悟性：${sc.comprehension}
   146→   146→灵石：${sc.spiritStones}  声望：${sc.reputation}

   148→   148→当前状态词条：
   149→   149→${statusList}
   150→   150→背包（储物袋，物品 id 已标注）：
   151→   151→${invList}
   152→   152→已装备（槽位/物品 id 已标注）：
   153→   153→${eqList}
   154→   154→
   155→   155→【记忆检索区】长期记忆：
   156→   156→${memory}
   157→   157→
   158→   158→【短期对话区】最近事件：
   159→   159→${recentEvts}
   160→   160→
   161→   161→${isFateNode && ctx.nextFateNode ? `【命节点触发】本轮应触发命节点 #${ctx.nextFateNode.index}「${ctx.nextFateNode.name}」！
   162→   162→此命节点对应境界：${ctx.nextFateNode.realm}

   187→   187→removedItemIds：若事件中某物品被破坏/消耗/丢失（如战斗中兵器损毁、丹药被服用、法宝碎裂），把该物品的 id 填入此数组，引擎会自动从储物袋或已装备中移除并反向结算属性。无则留空数组。
   188→   188→

   192→   192→【物品生成规则——必须严格遵守】
   193→   193→- item_type 取值：weapon(兵器)/armor(防具)/accessory(饰物)/artifact(法宝)/consumable(丹药)/material(材料)/tool(器具)/scripture(功法)
   194→   194→- rarity 必须与玩家境界匹配，严禁越级给高级物品：
   195→   195→  * 凡人/炼气期：common 或 uncommon
   196→   196→  * 筑基期：uncommon 或 rare
   197→   197→  * 金丹期：rare 或 epic
   198→   198→  * 元婴期：epic 或 legendary
   199→   199→  * 化神及以上：legendary 或 mythic
   200→   200→- effects 必须给出且符合物品类型语义：
   201→   201→  * weapon(兵器)：effects 用 add operation，target_attribute 为 attack（如 +10 attack）；高阶兵器可加 speed 或 hp
   202→   202→  * armor(防具)：effects 用 add，target 为 defense（如 +8 defense）；高阶可加 maxHp
   203→   203→  * accessory(饰物)：effects 用 add，target 为 luck/comprehension/maxMp 等
   204→   204→  * artifact(法宝)：effects 用 add 或 multiply，target 为 attack/defense/speed/cultivationExp 等；高阶法宝可有 multiply 效果
   205→   205→  * consumable(丹药)：effects 用 add，target 为 hp/mp/cultivationExp/lifespan 等；服用后消失
   206→   206→  * scripture(功法)：effects 必含一条 multiply cultivationExp（修炼倍率，凡品×1.2~1.5、良品×1.5~2.0、稀有×2.0~3.0、史诗×3.0~4.0、传说×4.0~5.0、神话×5.0~6.0）；可附带 add cultivationExp 等被动
   207→   207→  * material/tool：通常无 effects，仅作剧情道具
   208→   208→- id 格式：item_<类型缩写>_<4位随机>，如 item_wpn_a3f2、item_scr_b8c1、item_pil_d2e4。同一事件多个物品 id 不可重复。
   209→   209→- name：符合修仙世界风格（如"青锋剑""玄铁甲""聚气丹""引气诀""紫金葫芦"），≤8字
   210→   210→- description：10-40字，描述外观/功效/来历
   211→   211→- source：必须填写，简述获得方式（如"宗门发放""秘境拾得""炼丹炉出""战胜妖兽所得"）
   212→   212→
   213→   213→【物品修改规则——AI 联动】
   214→   214→- 战斗中兵器/防具/法宝可能损毁：把对应物品 id 填入 removedItemIds
   215→   215→- 丹药服用消耗：填入 removedItemIds
   216→   216→- 物品升级/精炼：不要在 newItems 重复给已有物品；若要升级，用 removedItemIds 移除旧物品 + newItems 给出新版（同名但属性更强、rarity 更高）
   217→   217→- 偷窃/赠送/典当物品：用 removedItemIds 移除
   218→   218→- 新获物品：填入 newItems，必须含完整字段与 effects
   219→   219→
   220→   220→【奇缘异宝——特殊状态词条生成规则】
   221→   221→玩家「宝」页除装备与储物袋外，还有「奇缘异宝」栏，展示 category=special 或 identity 的状态词条（灵宠、命格、天赋、身份、特殊体质等）。这些词条会随剧情获得或失去，AI 可通过 newStatuses 联动修改：
   222→   222→- 灵宠/坐骑：如获灵宠，newStatuses 给出 {category:"special", name:"灵宠·小白", description:"一只通体雪白的灵狐，善感知", rarity:"rare", duration:-1, source:"山林拾得", effects:[{target_attribute:"luck",operation:"add",value:5,description:"灵宠伴身，气运微增"}]}

   310→   310→async function callLLM(systemPrompt: string, userPrompt: string, scenePrompt: string): Promise<any> {
   311→   311→  const zai = await getZAI();
   312→   312→  const fullSystem = `${systemPrompt}\n\n${scenePrompt}`;
   313→   313→  try {
   314→   314→    const completion = await zai.chat.completions.create({
   315→   315→      messages: [
   316→   316→        { role: 'system', content: fullSystem },
   317→   317→        { role: 'user', content: userPrompt },
   318→   318→      ],
   319→   319→      thinking: { type: 'disabled' },
   320→   320→    });
   321→   321→    const content = completion.choices[0]?.message?.content || '';
   322→   322→    return parseJSON(content);
   323→   323→  } catch (err: any) {
   324→   324→    console.error('LLM call failed:', err?.message || err);
   325→   325→    throw err;
   326→   326→  }
   327→   327→}
   328→   328→
   329→   329→// 从 LLM 输出中提取 JSON（兼容 ```json ``` 包装、未转义字符、尾随逗号等常见问题）
   330→   330→function parseJSON(content: string): any {
   331→   331→  let s = content.trim();
   332→   332→  // 移除 markdown 代码块
   333→   333→  const fenceMatch = s.match(/```(?:json)?\s*([\s\S]*?)```/);
   334→   334→  if (fenceMatch) {
   335→   335→    s = fenceMatch[1].trim();
   336→   336→  }
   337→   337→  // 找第一个 { 到最后一个 }
   338→   338→  const start = s.indexOf('{');
   339→   339→  const end = s.lastIndexOf('}');
   340→   340→  if (start >= 0 && end > start) {
   341→   341→    s = s.slice(start, end + 1);
   342→   342→  }
   343→   343→  // 尝试直接解析
   344→   344→  try {
   345→   345→    return JSON.parse(s);
   346→   346→  } catch {
   347→   347→    // 失败则尝试修复常见问题
   348→   348→    return JSON.parse(repairJSON(s));
   349→   349→  }
   350→   350→}
   351→   351→
   352→   352→// 修复 LLM 输出 JSON 的常见问题：
   353→   353→// 1. 字符串值内未转义的双引号（如 narrative: "他说"你好"了"）
   354→   354→// 2. 字符串值内的裸换行符（JSON 标准要求 \n）
   355→   355→// 3. 尾随逗号
   356→   356→// 策略：逐字符扫描，仅在字符串外应用结构修复，字符串内转义裸引号/换行
   357→   357→function repairJSON(s: string): string {
   358→   358→  let out = '';
   359→   359→  let inStr = false;
   360→   360→  let escape = false;
   361→   361→  for (let i = 0; i < s.length; i++) {
   362→   362→    const ch = s[i];
   363→   363→    if (inStr) {
   364→   364→      if (escape) {
   365→   365→        out += ch;
   366→   366→        escape = false;
   367→   367→        continue;
   368→   368→      }
   369→   369→      if (ch === '\\') {
   370→   370→        out += ch;
   371→   371→        escape = true;
   372→   372→        continue;
   373→   373→      }
   374→   374→      if (ch === '"') {
   375→   375→        // 判断这是字符串结束还是裸引号：
   376→   376→        // 看后面（跳过空格）是否是 , } ] : —— 若是则视为字符串结束，否则视为裸引号需转义
   377→   377→        let j = i + 1;
   378→   378→        while (j < s.length && (s[j] === ' ' || s[j] === '\t')) j++;
   379→   379→        const nextCh = s[j];
   380→   380→        if (nextCh === ',' || nextCh === '}' || nextCh === ']' || nextCh === ':' || nextCh === undefined) {
   381→   381→          out += ch;
   382→   382→          inStr = false;
   383→   383→        } else {
   384→   384→          // 裸引号，转义
   385→   385→          out += '\\"';
   386→   386→        }
   387→   387→        continue;
   388→   388→      }
   389→   389→      if (ch === '\n') {
   390→   390→        out += '\\n';
   391→   391→        continue;
   392→   392→      }
   393→   393→      if (ch === '\r') {
   394→   394→        out += '\\r';
   395→   395→        continue;
   396→   396→      }
   397→   397→      if (ch === '\t') {
   398→   398→        out += '\\t';
   399→   399→        continue;
   400→   400→      }
   401→   401→      out += ch;
   402→   402→    } else {
   403→   403→      if (ch === '"') {
   404→   404→        inStr = true;
   405→   405→        out += ch;
   406→   406→      } else {
   407→   407→        out += ch;
   408→   408→      }
   409→   409→    }
   410→   410→  }
   411→   411→  // 移除尾随逗号（,} 或 ,]）
   412→   412→  out = out.replace(/,(\s*[}\]])/g, '$1');
   413→   413→  return out;
   414→   414→}

   416→   416→// ==================== 对外接口 ====================
   417→   417→
   418→   418→// 后处理：修正 narrative 中主角年龄数字（AI 偶发幻觉会把年龄写错，如3岁写成"四岁"）
   419→   419→// 策略：匹配紧邻主角名或句首的"数字+岁"，若数字≠正确年龄则替换为正确年龄（支持中文数字与阿拉伯数字）
   420→   420→const ZH_DIGIT: Record<string, number> = { 零:0,一:1,二:2,两:2,三:3,四:4,五:5,六:6,七:7,八:8,九:9,十:10 };
   421→   421→const ZH_AGE: Record<number, string> = { 0:'零',1:'一',2:'二',3:'三',4:'四',5:'五',6:'六',7:'七',8:'八',9:'九',10:'十' };
   422→   422→
   423→   423→// 中文数字（0-99）转 number，非中文数字返回 null
   424→   424→function zhAgeToNum(s: string): number | null {
   425→   425→  if (/^\d+$/.test(s)) return parseInt(s, 10);
   426→   426→  if (!/^[\u96f6\u4e00\u4e8c\u4e24\u4e09\u56db\u4e94\u516d\u4e03\u516b\u4e5d\u5341]+$/.test(s)) return null;
   427→   427→  if (s === '十') return 10;
   428→   428→  if (s.startsWith('十')) return 10 + (ZH_DIGIT[s[1]] ?? 0);
   429→   429→  if (s.endsWith('十')) return (ZH_DIGIT[s[0]] ?? 0) * 10;
   430→   430→  if (s.includes('十')) {
   431→   431→    const parts = s.split('十');
   432→   432→    return (ZH_DIGIT[parts[0]] ?? 0) * 10 + (ZH_DIGIT[parts[1]] ?? 0);
   433→   433→  }
   434→   434→  // 单字
   435→   435→  if (s.length === 1 && s in ZH_DIGIT) return ZH_DIGIT[s];
   436→   436→  return null;
   437→   437→}
   438→   438→
   439→   439→// number → 中文数字（0-99），用于替换文案
   440→   440→function numToZhAge(n: number): string {
   441→   441→  if (n in ZH_AGE) return ZH_AGE[n];
   442→   442→  if (n < 20) return '十' + (ZH_AGE[n - 10] ?? String(n - 10));
   443→   443→  const tens = Math.floor(n / 10);
   444→   444→  const ones = n % 10;
   445→   445→  return (ZH_AGE[tens] ?? String(tens)) + '十' + (ones ? (ZH_AGE[ones] ?? String(ones)) : '');
   446→   446→}

   448→   448→function fixNarrativeAge(narrative: string, correctAge: number, charName: string): string {
   449→   449→  if (!narrative) return narrative;
   450→   450→  let out = narrative;
   451→   451→  // 匹配"数字+岁+的+角色名"或"数字+岁+的+主角"等明确指代主角的模式
   452→   452→  // 中文数字或阿拉伯数字，0-99
   453→   453→  const numPat = '([0-9]+|[零一二三四五六七八九十]{1,3})';
   454→   454→  // 模式1："X岁的{name}" / "X岁那年{name}" / "X岁时，{name}"
   455→   455→  const re1 = new RegExp(`${numPat}岁(?:的|那年|时[，,]?)(.{0,2}?)${escapeRegExp(charName)}`, 'g');
   456→   456→  // 模式2：句首"X岁，"或"X岁的"开头（通常指代主角）
   457→   457→  const re2 = new RegExp(`^${numPat}岁(?:的|，|时|那年)`, 'g');
   458→   458→  // 模式3："{name}...X岁" 紧邻（中间部分用非捕获组，保证数字是第一个捕获组）
   459→   459→  const re3 = new RegExp(`${escapeRegExp(charName)}(?:[\\s\\S]{0,8}?)${numPat}岁`, 'g');
   460→   460→
   461→   461→  const replaceNum = (m: string, g1: string, ...rest: any[]) => {
   462→   462→    const num = zhAgeToNum(g1);
   463→   463→    if (num === null || num === correctAge) return m; // 无法解析或本来就对，不动
   464→   464→    if (num < 0 || num > 150) return m; // 明显是别的语境（如寿元、年份）
   465→   465→    // 替换为正确年龄（保持原格式：若原文是阿拉伯数字就用阿拉伯，否则用中文）
   466→   466→    const isArabic = /[0-9]/.test(g1);
   467→   467→    const replacement = isArabic ? String(correctAge) : numToZhAge(correctAge);
   468→   468→    return m.replace(g1, replacement);
   469→   469→  };
   470→   470→
   471→   471→  out = out.replace(re1, replaceNum);
   472→   472→  out = out.replace(re2, replaceNum);
   473→   473→  out = out.replace(re3, replaceNum);
   474→   474→  return out;
   475→   475→}
   476→   476→
   477→   477→function escapeRegExp(s: string): string {
   478→   478→  return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
   479→   479→}

   574→   574→    removedItemIds: Array.isArray(raw?.removedItemIds) ? raw.removedItemIds.map((x: any) => String(x)).filter(Boolean) : [],
   575→   575→    memory: String(raw?.memory || ''),
   576→   576→    causedDeath: Boolean(raw?.causedDeath),
   577→   577→    deathReason: raw?.deathReason ? String(raw.deathReason) : undefined,
   578→   578→  };
   579→   579→}

   581→   581→function sanitizeInterfereOutput(raw: any): InterfereOutput {
   582→   582→  const cls: InputClass = ['action','dialogue','overreach','rule_manipulation'].includes(raw?.classification)
   583→   583→    ? raw.classification : 'action';
   584→   584→  const accepted = raw?.accepted !== false && cls !== 'overreach' && cls !== 'rule_manipulation';
   585→   585→
   586→   586→  const changes: AttributeChange[] = accepted && Array.isArray(raw?.changes)
   587→   587→    ? raw.changes.map((c: any) => ({
   588→   588→        attribute: String(c.attribute || ''),
   589→   589→        delta: Number(c.delta) || 0,
   590→   590→        reason: String(c.reason || ''),
   591→   591→      })).filter((c: AttributeChange) => c.attribute)
   592→   592→    : [];
   593→   593→
   594→   594→  const { statuses, items } = ensureUniqueIds(
   595→   595→    accepted && Array.isArray(raw?.newStatuses) ? raw.newStatuses : [],
   596→   596→    accepted && Array.isArray(raw?.newItems) ? raw.newItems : []
   597→   597→  );
   598→   598→
   599→   599→  return {
   600→   600→    classification: cls,
   601→   601→    accepted,
   602→   602→    narrative: String(raw?.narrative || (accepted ? '天道如是。' : '世界自按其轨运行。')),
   603→   603→    changes,
   604→   604→    newStatuses: statuses,
   605→   605→    newItems: items,
   606→   606→    removedItemIds: accepted && Array.isArray(raw?.removedItemIds) ? raw.removedItemIds.map((x: any) => String(x)).filter(Boolean) : [],
   607→   607→    memory: accepted ? String(raw?.memory || '') : '',
   608→   608→    ageAdvance: accepted ? Math.max(0, Math.min(5, Number(raw?.ageAdvance) || 0)) : 0,
   609→   609→  };
   610→   610→}

   614→   614→// 引擎权威：灵根类型与五行组合由后端按概率随机生成（LLM 不可自由发挥，避免每次重生结果趋同）
   615→   615→// 灵根类型按 rarity 权重抽取（none30/mixed25/common20/pure15/heavenly8/chaos2）
   616→   616→
   617→   617→const ALL_ELEMENTS: Element[] = ['metal', 'wood', 'water', 'fire', 'earth'];
   618→   618→
   619→   619→function rollSpiritualRoot(): SpiritualRoot {
   620→   620→  const entries = Object.entries(SPIRITUAL_ROOTS) as [SpiritualRoot, { rarity: number }][];
   621→   621→  const total = entries.reduce((s, [, v]) => s + v.rarity, 0);
   622→   622→  let r = Math.random() * total;
   623→   623→  for (const [k, v] of entries) {
   624→   624→    r -= v.rarity;
   625→   625→    if (r <= 0) return k;
   626→   626→  }
   627→   627→  return 'mixed';
   628→   628→}
   629→   629→
   630→   630→// 根据灵根类型随机生成五行倾向（哪些元素突出），并返回初始五行数值
   631→   631→// 返回 { elements, picked }：picked 为本次突出的元素列表（用于让 LLM 生成 rootDetail）
   632→   632→function rollElements(root: SpiritualRoot): { elements: Record<Element, number>; picked: Element[] } {
   633→   633→  const shuffle = <T,>(arr: T[]): T[] => {
   634→   634→    const a = [...arr];
   635→   635→    for (let i = a.length - 1; i > 0; i--) {
   636→   636→      const j = Math.floor(Math.random() * (i + 1));
   637→   637→      [a[i], a[j]] = [a[j], a[i]];
   638→   638→    }
   639→   639→    return a;
   640→   640→  };
   641→   641→
   642→   642→  switch (root) {
   643→   643→    case 'none':
   644→   644→      // 无灵根：五行都很低
   645→   645→      return { elements: { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 }, picked: [] };
   646→   646→    case 'mixed':
   647→   647→      // 杂灵根：五行皆中等
   648→   648→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   649→   649→    case 'common': {
   650→   650→      // 凡灵根：2-3 种属性突出
   651→   651→      const count = 2 + Math.floor(Math.random() * 2); // 2 或 3
   652→   652→      const picked = shuffle(ALL_ELEMENTS).slice(0, count);
   653→   653→      const elements: Record<Element, number> = { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 };
   654→   654→      for (const e of picked) elements[e] = 30 + Math.floor(Math.random() * 11); // 30-40
   655→   655→      return { elements, picked };
   656→   656→    }
   657→   657→    case 'pure': {
   658→   658→      // 真灵根：单属性突出
   659→   659→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   660→   660→      const elements: Record<Element, number> = { metal: 5, wood: 5, water: 5, fire: 5, earth: 5 };
   661→   661→      elements[picked[0]] = 50 + Math.floor(Math.random() * 11); // 50-60
   662→   662→      return { elements, picked };
   663→   663→    }
   664→   664→    case 'heavenly': {
   665→   665→      // 天灵根：单属性极突出
   666→   666→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   667→   667→      const elements: Record<Element, number> = { metal: 3, wood: 3, water: 3, fire: 3, earth: 3 };
   668→   668→      elements[picked[0]] = 70 + Math.floor(Math.random() * 11); // 70-80
   669→   669→      return { elements, picked };
   670→   670→    }
   671→   671→    case 'chaos':
   672→   672→      // 混沌灵根：五行皆高
   673→   673→      return { elements: { metal: 45, wood: 45, water: 45, fire: 45, earth: 45 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   674→   674→    default:
   675→   675→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: [] };
   676→   676→  }
   677→   677→}

   681→   681→  return els.map(e => ELEMENTS[e].name).join('');
