   154→function buildAdvancePrompt(ctx: EngineStateContext, isFateNode: boolean): string {
   155→  const sc = ctx.character;
   156→  const elements = `金${sc.elements.metal}/木${sc.elements.wood}/水${sc.elements.water}/火${sc.elements.fire}/土${sc.elements.earth}`;
   157→  const statusList = ctx.activeStatuses.length
   158→    ? ctx.activeStatuses.map(s => `- ${s.name}（${s.category}，${s.rarity}）：${s.description}`).join('\n')
   159→    : '无';
   160→  const invList = ctx.inventory.length
   161→    ? ctx.inventory.map(i => `- [id:${i.id}] ${i.name}（${i.rarity}/${i.item_type}）：${i.description}${i.equipNote ? `；装备位置：${i.equipNote}` : ''}${i.effects?.length ? '；效果：' + i.effects.map(e => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   162→    : '无';
   163→  // 已装备：数组展示，不再有固定槽位（玩家可戴多枚戒指、脖挂一串储物戒指等）
   164→  const eqArr = Array.isArray(ctx.equipped) ? ctx.equipped : [];
   165→  const eqList = eqArr.length
   166→    ? eqArr.map((it: any) => `- [id:${it.id}] ${it.name}（${it.rarity}/${it.item_type}）${it.equipNote ? `·${it.equipNote}` : ''}：${it.description}${it.effects?.length ? '；效果：' + it.effects.map((e: any) => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   167→    : '无';
   168→  const recentEvts = ctx.recentEvents.length
   169→    ? ctx.recentEvents.map(e => `${e.age}岁：${e.title}——${e.narrative.slice(0, 80)}`).join('\n')
   170→    : '无';
   171→  const memory = ctx.longTermMemory.length
   172→    ? ctx.longTermMemory.map(m => `- ${m}`).join('\n')
   173→    : '无';
   174→  const mult = ctx.cultivationMultiplier || 0;
   175→  const multDesc = mult > 0 ? `${mult.toFixed(2)}倍（已含灵根与功法加成）` : '0（无灵根，无法修炼）';
   176→  const curInsight = ctx.cultivationInsight || '';
   177→  // 引擎权威计算的来源条目（灵根 + 已装备功法 + 状态词条中的 cultivationExp 效果）
   178→  // 这些数字是准确的，与顶部倍率一致；AI 必须在 cultivationInsight 文本中引用这些准确数字
   179→  const engineFactors = (ctx.cultivationFactors && ctx.cultivationFactors.length)
   180→    ? ctx.cultivationFactors.map(f => `${f.name}(${f.operation === 'multiply' ? '×' : '+'}${f.value}${f.note ? '，' + f.note : ''})`).join('，')
   181→    : '（暂无来源——可能无灵根或未装备功法）';
   182→  // 储物袋容量信息
   183→  const storageCap = ctx.storageCapacity ?? 5;
   184→  const invCount = ctx.inventory.length;
   185→  const hasBag = invCount > 0 && ctx.inventory.some(i => i.item_type === 'tool' && (i.effects || []).some(e => e.target_attribute === 'storageCapacity'));
   186→  const storageDesc = `${invCount}/${storageCap}件${hasBag ? '（已有储物袋）' : '（无储物袋，上限仅 5 件）'}`;
   187→
   188→  return `【状态快照区】
   189→角色：${sc.name}（${sc.gender === 'male' ? '男' : '女'}），${sc.age}岁
   190→寿元：${sc.lifespan}岁（剩余约${sc.lifespan - sc.age}岁）
   191→灵根：${sc.rootDetail || sc.spiritualRoot}
   192→境界：${sc.realmName}${sc.realmLevel > 0 ? `（${sc.realmLevel}层）` : ''}
   193→修为：${sc.cultivationExp}/${sc.expToBreak}（修炼速度：${multDesc}，你给出的 cultivationExp 正向增量会被该倍率放大）
   194→五行倾向：${elements}
   195→生命：${sc.hp}/${sc.maxHp}  灵力：${sc.mp}/${sc.maxMp}
   196→攻击：${sc.attack}  防御：${sc.defense}  速度：${sc.speed}
   197→气运：${sc.luck}  悟性：${sc.comprehension}
   198→灵石：${sc.spiritStones}  声望：${sc.reputation}
   199→宗门：${sc.faction || '散修'}  师承：${sc.master || '无'}  所在：${sc.location}
   200→当前状态词条：
   201→${statusList}
   202→背包（${storageDesc}，物品 id 已标注）：
   203→${invList}
   204→已装备（数组，无槽位上限，物品 id 已标注）：
   205→${eqList}
   206→
   207→【当前修炼心得】（玩家「宝」页修炼速度栏展示文本，由你上一轮生成，本轮可更新）
   208→${curInsight || '（尚未生成，本轮请首次生成）'}
   209→【当前修炼速度来源条目】（引擎权威计算，数字准确，与顶部倍率一致；你必须在 cultivationInsight 文本中引用这些来源与数字，不可编造或增减）
   210→${engineFactors}
   211→
   212→【事件蓝图区】（本轮事件必须围绕此主题展开——天道抽取，你不可更改）
   213→主题：${ctx.blueprint ? `${ctx.blueprint.name}（分类：${ctx.blueprint.category}）` : '无（自由发挥，但须避免与最近事件类型重复）'}
   214→${ctx.blueprint ? `描述：${ctx.blueprint.description}` : ''}
   215→${ctx.blueprint?.examples?.length ? `灵感参考（不可照抄，需融入角色处境）：${ctx.blueprint.examples.join('；')}` : ''}
   216→
   217→【角色主动意图区】（角色当前会主动做这些事——你必须在 narrative 中体现）
   218→${ctx.characterIntents && ctx.characterIntents.length
   219→  ? ctx.characterIntents.map(i => `- [优先级${i.priority}] ${i.title}：${i.description}`).join('\n')
   220→  : '（无特定主动意图，按蓝图主题自由生成）'}
   221→
   222→【未决线索区】（必须保持连续性！urgent 的本轮必须推进或解决）
   223→${ctx.pendingThreads && ctx.pendingThreads.length
   224→  ? ctx.pendingThreads.map(t => `- [id:${t.id}][${t.status}] ${t.title}（截止 ${t.deadlineAge} 岁，剩 ${t.deadlineAge - sc.age} 岁，进度 ${t.progress}%）：${t.description}${t.reward ? `；奖励：${t.reward}` : ''}${t.failureCost ? `；失败代价：${t.failureCost}` : ''}`).join('\n')
   225→  : '（无未决线索）'}
   226→
   227→【反重复机制】
   228→最近事件类型（严禁连续 3 次同类）：${ctx.recentEventTypes && ctx.recentEventTypes.length ? ctx.recentEventTypes.join(' → ') : '无'}
   229→最近蓝图分类（避免连续同类）：${ctx.recentBlueprintCategories && ctx.recentBlueprintCategories.length ? ctx.recentBlueprintCategories.join(' → ') : '无'}
   230→
   231→【记忆检索区】长期记忆：
   232→${memory}
   233→
   234→【短期对话区】最近事件：
   235→${recentEvts}
   236→
   237→${isFateNode && ctx.nextFateNode ? `【命节点触发】本轮应触发命节点 #${ctx.nextFateNode.index}「${ctx.nextFateNode.name}」！
   238→此命节点对应境界：${ctx.nextFateNode.realm}
   239→本命节点必须 hasChoice=true，生成 3-4 个有意义的选项供玩家选择。
   240→narrative 应铺垫此命节点的核心冲突，choice.prompt 描述抉择情境。
   241→此命节点完成后角色将永久性改变（获得身份、师承、命格等）。` : '【命节点】本轮无命节点触发，生成普通年龄事件。'}
   242→
   243→请生成 JSON，schema 如下：
   244→{
   245→  "title": "事件标题（≤16字）",
   246→  "narrative": "叙事正文（100-250字，命节点250-500字）",
   247→  "eventType": "normal | fate_node | choice | combat | breakthrough | death | ascension",

   321→  * armor(防具)：effects 用 add，target 为 defense（如 +8 defense）；高阶可加 maxHp
   322→  * accessory(饰物)：effects 用 add，target 为 luck/comprehension/maxMp 等
   323→  * artifact(法宝)：effects 用 add 或 multiply，target 为 attack/defense/speed/cultivationExp 等；高阶法宝可有 multiply 效果

   331→- source：必须填写，简述获得方式（如"宗门发放""秘境拾得""炼丹炉出""战胜妖兽所得"）
   332→- equipNote（可选）：若该物品默认就该戴在特定位置，可给出（如"左手""脖挂""腰悬"）；玩家装备时若无则按类型默认生成

   336→- 丹药服用消耗：填入 removedItemIds

   385→- 严禁出现 JSON 转义问题：文本内不得出现裸双引号、裸换行符；书名号《》可以用；若需引用功法名，直接写书名即可`;

   503→async function callLLM(systemPrompt: string, userPrompt: string, scenePrompt: string): Promise<any> {
   504→  const zai = await getZAI();
   505→  const fullSystem = `${systemPrompt}\n\n${scenePrompt}`;
   506→  try {
   507→    const completion = await zai.chat.completions.create({
   508→      messages: [
   509→        { role: 'system', content: fullSystem },
   510→        { role: 'user', content: userPrompt },
   511→      ],
   512→      thinking: { type: 'disabled' },
   513→    });
   514→    const content = completion.choices[0]?.message?.content || '';
   515→    return parseJSON(content);
   516→  } catch (err: any) {
   517→    console.error('LLM call failed:', err?.message || err);
   518→    throw err;
   519→  }
   520→}

   522→// 从 LLM 输出中提取 JSON（兼容 ```json ``` 包装、未转义字符、尾随逗号等常见问题）
   523→function parseJSON(content: string): any {
   524→  let s = content.trim();
   525→  // 移除 markdown 代码块
   526→  const fenceMatch = s.match(/```(?:json)?\s*([\s\S]*?)```/);
   527→  if (fenceMatch) {
   528→    s = fenceMatch[1].trim();
   529→  }
   530→  // 找第一个 { 到最后一个 }
   531→  const start = s.indexOf('{');
   532→  const end = s.lastIndexOf('}');
   533→  if (start >= 0 && end > start) {
   534→    s = s.slice(start, end + 1);
   535→  }
   536→  // 尝试直接解析
   537→  try {
   538→    return JSON.parse(s);
   539→  } catch {
   540→    // 失败则尝试修复常见问题
   541→    return JSON.parse(repairJSON(s));
   542→  }
   543→}
   544→
   545→// 修复 LLM 输出 JSON 的常见问题：
   546→// 1. 字符串值内未转义的双引号（如 narrative: "他说"你好"了"）
   547→// 2. 字符串值内的裸换行符（JSON 标准要求 \n）
   548→// 3. 尾随逗号
   549→// 策略：逐字符扫描，仅在字符串外应用结构修复，字符串内转义裸引号/换行
   550→function repairJSON(s: string): string {
   551→  let out = '';
   552→  let inStr = false;
   553→  let escape = false;
   554→  for (let i = 0; i < s.length; i++) {
   555→    const ch = s[i];
   556→    if (inStr) {
   557→      if (escape) {
   558→        out += ch;
   559→        escape = false;
   560→        continue;
   561→      }
   562→      if (ch === '\\') {
   563→        out += ch;
   564→        escape = true;
   565→        continue;
   566→      }
   567→      if (ch === '"') {
   568→        // 判断这是字符串结束还是裸引号：
   569→        // 看后面（跳过空格）是否是 , } ] : —— 若是则视为字符串结束，否则视为裸引号需转义
   570→        let j = i + 1;
   571→        while (j < s.length && (s[j] === ' ' || s[j] === '\t')) j++;
   572→        const nextCh = s[j];
   573→        if (nextCh === ',' || nextCh === '}' || nextCh === ']' || nextCh === ':' || nextCh === undefined) {
   574→          out += ch;
   575→          inStr = false;
   576→        } else {
   577→          // 裸引号，转义
   578→          out += '\\"';
   579→        }
   580→        continue;
   581→      }
   582→      if (ch === '\n') {
   583→        out += '\\n';
   584→        continue;
   585→      }
   586→      if (ch === '\r') {
   587→        out += '\\r';
   588→        continue;
   589→      }
   590→      if (ch === '\t') {
   591→        out += '\\t';
   592→        continue;
   593→      }
   594→      out += ch;
   595→    } else {
   596→      if (ch === '"') {
   597→        inStr = true;
   598→        out += ch;
   599→      } else {
   600→        out += ch;
   601→      }
   602→    }
   603→  }
   604→  // 移除尾随逗号（,} 或 ,]）
   605→  out = out.replace(/,(\s*[}\]])/g, '$1');
   606→  return out;
   607→}

   611→// 后处理：修正 narrative 中主角年龄数字（AI 偶发幻觉会把年龄写错，如3岁写成"四岁"）
   612→// 策略：匹配紧邻主角名或句首的"数字+岁"，若数字≠正确年龄则替换为正确年龄（支持中文数字与阿拉伯数字）
   613→const ZH_DIGIT: Record<string, number> = { 零:0,一:1,二:2,两:2,三:3,四:4,五:5,六:6,七:7,八:8,九:9,十:10 };
   614→const ZH_AGE: Record<number, string> = { 0:'零',1:'一',2:'二',3:'三',4:'四',5:'五',6:'六',7:'七',8:'八',9:'九',10:'十' };
   615→
   616→// 中文数字（0-99）转 number，非中文数字返回 null
   617→function zhAgeToNum(s: string): number | null {
   618→  if (/^\d+$/.test(s)) return parseInt(s, 10);
   619→  if (!/^[\u96f6\u4e00\u4e8c\u4e24\u4e09\u56db\u4e94\u516d\u4e03\u516b\u4e5d\u5341]+$/.test(s)) return null;
   620→  if (s === '十') return 10;
   621→  if (s.startsWith('十')) return 10 + (ZH_DIGIT[s[1]] ?? 0);
   622→  if (s.endsWith('十')) return (ZH_DIGIT[s[0]] ?? 0) * 10;
   623→  if (s.includes('十')) {
   624→    const parts = s.split('十');
   625→    return (ZH_DIGIT[parts[0]] ?? 0) * 10 + (ZH_DIGIT[parts[1]] ?? 0);
   626→  }
   627→  // 单字
   628→  if (s.length === 1 && s in ZH_DIGIT) return ZH_DIGIT[s];
   629→  return null;
   630→}
   631→
   632→// number → 中文数字（0-99），用于替换文案
   633→function numToZhAge(n: number): string {
   634→  if (n in ZH_AGE) return ZH_AGE[n];
   635→  if (n < 20) return '十' + (ZH_AGE[n - 10] ?? String(n - 10));
   636→  const tens = Math.floor(n / 10);
   637→  const ones = n % 10;
   638→  return (ZH_AGE[tens] ?? String(tens)) + '十' + (ones ? (ZH_AGE[ones] ?? String(ones)) : '');
   639→}

   641→function fixNarrativeAge(narrative: string, correctAge: number, charName: string): string {
   642→  if (!narrative) return narrative;
   643→  let out = narrative;
   644→  // 匹配"数字+岁+的+角色名"或"数字+岁+的+主角"等明确指代主角的模式
   645→  // 中文数字或阿拉伯数字，0-99
   646→  const numPat = '([0-9]+|[零一二三四五六七八九十]{1,3})';
   647→  // 模式1："X岁的{name}" / "X岁那年{name}" / "X岁时，{name}"
   648→  const re1 = new RegExp(`${numPat}岁(?:的|那年|时[，,]?)(.{0,2}?)${escapeRegExp(charName)}`, 'g');
   649→  // 模式2：句首"X岁，"或"X岁的"开头（通常指代主角）
   650→  const re2 = new RegExp(`^${numPat}岁(?:的|，|时|那年)`, 'g');
   651→  // 模式3："{name}...X岁" 紧邻（中间部分用非捕获组，保证数字是第一个捕获组）
   652→  const re3 = new RegExp(`${escapeRegExp(charName)}(?:[\\s\\S]{0,8}?)${numPat}岁`, 'g');
   653→
   654→  const replaceNum = (m: string, g1: string, ...rest: any[]) => {
   655→    const num = zhAgeToNum(g1);
   656→    if (num === null || num === correctAge) return m; // 无法解析或本来就对，不动
   657→    if (num < 0 || num > 150) return m; // 明显是别的语境（如寿元、年份）
   658→    // 替换为正确年龄（保持原格式：若原文是阿拉伯数字就用阿拉伯，否则用中文）
   659→    const isArabic = /[0-9]/.test(g1);
   660→    const replacement = isArabic ? String(correctAge) : numToZhAge(correctAge);
   661→    return m.replace(g1, replacement);
   662→  };
   663→
   664→  out = out.replace(re1, replaceNum);
   665→  out = out.replace(re2, replaceNum);
   666→  out = out.replace(re3, replaceNum);
   667→  return out;
   668→}
   669→
   670→function escapeRegExp(s: string): string {
   671→  return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
   672→}

   807→function sanitizeEventOutput(raw: any): AIEventOutput {
   808→  const changes: AttributeChange[] = Array.isArray(raw?.changes) ? raw.changes.map((c: any) => ({
   809→    attribute: String(c.attribute || ''),
   810→    delta: Number(c.delta) || 0,
   811→    reason: String(c.reason || ''),
   812→  })).filter((c: AttributeChange) => c.attribute) : [];

   937→// 引擎权威：灵根类型与五行组合由后端按概率随机生成（LLM 不可自由发挥，避免每次重生结果趋同）
   938→// 灵根类型按 rarity 权重抽取（none30/mixed25/common20/pure15/heavenly8/chaos2）
   939→
   940→const ALL_ELEMENTS: Element[] = ['metal', 'wood', 'water', 'fire', 'earth'];
   941→
   942→function rollSpiritualRoot(): SpiritualRoot {
   943→  const entries = Object.entries(SPIRITUAL_ROOTS) as [SpiritualRoot, { rarity: number }][];
   944→  const total = entries.reduce((s, [, v]) => s + v.rarity, 0);
   945→  let r = Math.random() * total;
   946→  for (const [k, v] of entries) {
   947→    r -= v.rarity;
   948→    if (r <= 0) return k;
   949→  }
   950→  return 'mixed';
   951→}
   952→
   953→// 根据灵根类型随机生成五行倾向（哪些元素突出），并返回初始五行数值
   954→// 返回 { elements, picked }：picked 为本次突出的元素列表（用于让 LLM 生成 rootDetail）
   955→function rollElements(root: SpiritualRoot): { elements: Record<Element, number>; picked: Element[] } {
   956→  const shuffle = <T,>(arr: T[]): T[] => {
   957→    const a = [...arr];
   958→    for (let i = a.length - 1; i > 0; i--) {
   959→      const j = Math.floor(Math.random() * (i + 1));
   960→      [a[i], a[j]] = [a[j], a[i]];
   961→    }
   962→    return a;
   963→  };
   964→
   965→  switch (root) {
   966→    case 'none':
   967→      // 无灵根：五行都很低
   968→      return { elements: { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 }, picked: [] };
   969→    case 'mixed':
   970→      // 杂灵根：五行皆中等
   971→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   972→    case 'common': {
   973→      // 凡灵根：2-3 种属性突出
   974→      const count = 2 + Math.floor(Math.random() * 2); // 2 或 3
   975→      const picked = shuffle(ALL_ELEMENTS).slice(0, count);
   976→      const elements: Record<Element, number> = { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 };
   977→      for (const e of picked) elements[e] = 30 + Math.floor(Math.random() * 11); // 30-40
   978→      return { elements, picked };
   979→    }
   980→    case 'pure': {
   981→      // 真灵根：单属性突出
   982→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   983→      const elements: Record<Element, number> = { metal: 5, wood: 5, water: 5, fire: 5, earth: 5 };
   984→      elements[picked[0]] = 50 + Math.floor(Math.random() * 11); // 50-60
   985→      return { elements, picked };
   986→    }
   987→    case 'heavenly': {
   988→      // 天灵根：单属性极突出
   989→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   990→      const elements: Record<Element, number> = { metal: 3, wood: 3, water: 3, fire: 3, earth: 3 };
   991→      elements[picked[0]] = 70 + Math.floor(Math.random() * 11); // 70-80
   992→      return { elements, picked };
   993→    }
   994→    case 'chaos':
   995→      // 混沌灵根：五行皆高
   996→      return { elements: { metal: 45, wood: 45, water: 45, fire: 45, earth: 45 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   997→    default:
   998→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: [] };
   999→  }
  1000→}
