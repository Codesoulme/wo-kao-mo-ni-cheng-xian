   116→function buildAdvancePrompt(ctx: EngineStateContext, isFateNode: boolean): string {
   117→  const sc = ctx.character;
   118→  const elements = `金${sc.elements.metal}/木${sc.elements.wood}/水${sc.elements.water}/火${sc.elements.fire}/土${sc.elements.earth}`;
   119→  const statusList = ctx.activeStatuses.length
   120→    ? ctx.activeStatuses.map(s => `- ${s.name}（${s.category}，${s.rarity}）：${s.description}`).join('\n')
   121→    : '无';
   122→  const invList = ctx.inventory.length
   123→    ? ctx.inventory.map(i => `- [id:${i.id}] ${i.name}（${i.rarity}/${i.item_type}）：${i.description}${i.equipNote ? `；装备位置：${i.equipNote}` : ''}${i.effects?.length ? '；效果：' + i.effects.map(e => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   124→    : '无';
   125→  // 已装备：数组展示，不再有固定槽位（玩家可戴多枚戒指、脖挂一串储物戒指等）
   126→  const eqArr = Array.isArray(ctx.equipped) ? ctx.equipped : [];
   127→  const eqList = eqArr.length
   128→    ? eqArr.map((it: any) => `- [id:${it.id}] ${it.name}（${it.rarity}/${it.item_type}）${it.equipNote ? `·${it.equipNote}` : ''}：${it.description}${it.effects?.length ? '；效果：' + it.effects.map((e: any) => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   129→    : '无';
   130→  const recentEvts = ctx.recentEvents.length
   131→    ? ctx.recentEvents.map(e => `${e.age}岁：${e.title}——${e.narrative.slice(0, 80)}`).join('\n')
   132→    : '无';
   133→  const memory = ctx.longTermMemory.length
   134→    ? ctx.longTermMemory.map(m => `- ${m}`).join('\n')
   135→    : '无';
   136→  const mult = ctx.cultivationMultiplier || 0;
   137→  const multDesc = mult > 0 ? `${mult.toFixed(2)}倍（已含灵根与功法加成）` : '0（无灵根，无法修炼）';
   138→  const curInsight = ctx.cultivationInsight || '';
   139→  // 引擎权威计算的来源条目（灵根 + 已装备功法 + 状态词条中的 cultivationExp 效果）
   140→  // 这些数字是准确的，与顶部倍率一致；AI 必须在 cultivationInsight 文本中引用这些准确数字
   141→  const engineFactors = (ctx.cultivationFactors && ctx.cultivationFactors.length)
   142→    ? ctx.cultivationFactors.map(f => `${f.name}(${f.operation === 'multiply' ? '×' : '+'}${f.value}${f.note ? '，' + f.note : ''})`).join('，')
   143→    : '（暂无来源——可能无灵根或未装备功法）';
   144→  // 储物袋容量信息
   145→  const storageCap = ctx.storageCapacity ?? 5;
   146→  const invCount = ctx.inventory.length;
   147→  const hasBag = invCount > 0 && ctx.inventory.some(i => i.item_type === 'tool' && (i.effects || []).some(e => e.target_attribute === 'storageCapacity'));
   148→  const storageDesc = `${invCount}/${storageCap}件${hasBag ? '（已有储物袋）' : '（无储物袋，上限仅 5 件）'}`;
   149→
   150→  return `【状态快照区】
   151→角色：${sc.name}（${sc.gender === 'male' ? '男' : '女'}），${sc.age}岁
   152→寿元：${sc.lifespan}岁（剩余约${sc.lifespan - sc.age}岁）
   153→灵根：${sc.rootDetail || sc.spiritualRoot}
   154→境界：${sc.realmName}${sc.realmLevel > 0 ? `（${sc.realmLevel}层）` : ''}
   155→修为：${sc.cultivationExp}/${sc.expToBreak}（修炼速度：${multDesc}，你给出的 cultivationExp 正向增量会被该倍率放大）
   156→五行倾向：${elements}
   157→生命：${sc.hp}/${sc.maxHp}  灵力：${sc.mp}/${sc.maxMp}
   158→攻击：${sc.attack}  防御：${sc.defense}  速度：${sc.speed}
   159→气运：${sc.luck}  悟性：${sc.comprehension}
   160→灵石：${sc.spiritStones}  声望：${sc.reputation}
   161→宗门：${sc.faction || '散修'}  师承：${sc.master || '无'}  所在：${sc.location}
   162→当前状态词条：
   163→${statusList}
   164→背包（${storageDesc}，物品 id 已标注）：
   165→${invList}
   166→已装备（数组，无槽位上限，物品 id 已标注）：
   167→${eqList}
   168→
   169→【当前修炼心得】（玩家「宝」页修炼速度栏展示文本，由你上一轮生成，本轮可更新）
   170→${curInsight || '（尚未生成，本轮请首次生成）'}
   171→【当前修炼速度来源条目】（引擎权威计算，数字准确，与顶部倍率一致；你必须在 cultivationInsight 文本中引用这些来源与数字，不可编造或增减）
   172→${engineFactors}
   173→
   174→【记忆检索区】长期记忆：
   175→${memory}
   176→
   177→【短期对话区】最近事件：
   178→${recentEvts}
   179→
   180→${isFateNode && ctx.nextFateNode ? `【命节点触发】本轮应触发命节点 #${ctx.nextFateNode.index}「${ctx.nextFateNode.name}」！
   181→此命节点对应境界：${ctx.nextFateNode.realm}
   182→本命节点必须 hasChoice=true，生成 3-4 个有意义的选项供玩家选择。
   183→narrative 应铺垫此命节点的核心冲突，choice.prompt 描述抉择情境。
   184→此命节点完成后角色将永久性改变（获得身份、师承、命格等）。` : '【命节点】本轮无命节点触发，生成普通年龄事件。'}
   185→
   186→请生成 JSON，schema 如下：
   187→{
   188→  "title": "事件标题（≤16字）",
   189→  "narrative": "叙事正文（100-300字，命节点200-400字）",
   190→  "eventType": "normal | fate_node | choice | combat | breakthrough | death | ascension",
   191→  "changes": [{"attribute":"cultivationExp","delta":10,"reason":"修炼精进"}],
   192→  "newStatuses": [],
   193→  "newItems": [],
   194→  "removedItemIds": [],
   195→  "newEquippedItems": [],
   196→  "equipItemIds": [],
   197→  "unequipItemIds": [],
   198→  "memory": "本岁关键事件一句话摘要，写入长期记忆",
   199→  "cultivationInsight": "修炼心得文本（60-150字，见下方生成规则；必须引用引擎提供的准确来源名称与数字）",
   200→  "hasChoice": false,
   201→  "choice": null,
   202→  "triggeredBreakthrough": false,
   203→  "causedDeath": false,
   204→  "causedAscension": false
   205→}
   206→
   207→可修改属性白名单：${ctx.availableAttributes.join(', ')}
   208→注意：attribute 必须在白名单内；delta 合理（普通事件 -50~+100，奇遇 -200~+500）；newStatuses 与 newItems 给出完整字段。
   209→严禁修改 age（年龄由天道推进，每岁固定 +1，AI 不得在 changes 中包含 age）。
   210→removedItemIds：若事件中某物品被破坏/消耗/丢失（如战斗中兵器损毁、丹药被服用、法宝碎裂），把该物品的 id 填入此数组，引擎会自动从储物袋或已装备中移除并反向结算属性。无则留空数组。
   211→equipItemIds：若想把背包里已有的物品装备上去，把其 id 填入此数组（引擎自动从 inventory 移到 equipped）。
   212→unequipItemIds：若想把已装备的物品卸下来，把其 id 填入此数组（引擎自动从 equipped 移到 inventory）。
   213→newEquippedItems：用于「AI 创造性装备」场景——例如玩家说"我把一堆储物戒指用绳子串成项链戴在脖子上"，你认为合理，就把这些戒指（或合并后的项链条目）作为新物品放入此数组，并给出 equipNote 描述位置（如"项链·储物戒指×5"）。引擎会直接把它们放入 equipped，不占背包位置。物品本身可来自玩家描述的「已有物品的组合」或全新创造的合成物。
   214→
   215→statusEntry 结构：{id,name,description,category(attribute/skill/buff/debuff/special/identity/quest/environment),rarity(common/uncommon/rare/epic/legendary/mythic),duration(-1永久/正数为剩余岁数),source,effects:[{target_attribute,operation(add/multiply/override/cap/floor/trigger),value,description}]}
   216→
   217→itemEntry 结构：{id,name,description,item_type,rarity,effects:[...],source,equipNote(可选，装备位置自由文本)}
   218→【物品生成规则——必须严格遵守】
   219→- item_type 取值（必须严格使用以下之一）：weapon(兵器)/armor(防具)/accessory(饰物)/artifact(法宝)/consumable(丹药)/material(材料)/tool(器具)/scripture(功法)
   220→  * 储物袋必须用 item_type='tool'（不可用 'storage' 等其他值，会报错）
   221→- rarity 必须与玩家境界匹配，严禁越级给高级物品：
   222→  * 凡人/炼气期：common 或 uncommon
   223→  * 筑基期：uncommon 或 rare
   224→  * 金丹期：rare 或 epic
   225→  * 元婴期：epic 或 legendary
   226→  * 化神及以上：legendary 或 mythic
   227→- effects 必须给出且符合物品类型语义：
   228→  * weapon(兵器)：effects 用 add operation，target_attribute 为 attack（如 +10 attack）；高阶兵器可加 speed 或 hp
   229→  * armor(防具)：effects 用 add，target 为 defense（如 +8 defense）；高阶可加 maxHp
   230→  * accessory(饰物)：effects 用 add，target 为 luck/comprehension/maxMp 等
   231→  * artifact(法宝)：effects 用 add 或 multiply，target 为 attack/defense/speed/cultivationExp 等；高阶法宝可有 multiply 效果
   232→  * consumable(丹药)：effects 用 add，target 为 hp/mp/cultivationExp/lifespan 等；服用后消失
   233→  * scripture(功法)：effects 必含一条 multiply cultivationExp（修炼倍率，凡品×1.2~1.5、良品×1.5~2.0、稀有×2.0~3.0、史诗×3.0~4.0、传说×4.0~5.0、神话×5.0~6.0）；可附带 add cultivationExp 等被动
   234→  * tool(器具)：可以是储物袋——effects 用 add，target 为 storageCapacity（如 +10 storageCapacity 表示增加 10 格容量）；储物袋获得即扩容，无需装备
   235→  * material：通常无 effects，仅作剧情道具或炼丹材料
   236→- id 格式：item_<类型缩写>_<4位随机>，如 item_wpn_a3f2、item_scr_b8c1、item_pil_d2e4、item_bag_f0a1。同一事件多个物品 id 不可重复。
   237→- name：符合修仙世界风格（如"青锋剑""玄铁甲""聚气丹""引气诀""紫金葫芦""初级储物袋"），≤8字
   238→- description：10-40字，描述外观/功效/来历
   239→- source：必须填写，简述获得方式（如"宗门发放""秘境拾得""炼丹炉出""战胜妖兽所得"）
   240→- equipNote（可选）：若该物品默认就该戴在特定位置，可给出（如"左手""脖挂""腰悬"）；玩家装备时若无则按类型默认生成
   241→
   242→【物品修改规则——AI 联动】
   243→- 战斗中兵器/防具/法宝可能损毁：把对应物品 id 填入 removedItemIds
   244→- 丹药服用消耗：填入 removedItemIds
   245→- 物品升级/精炼：不要在 newItems 重复给已有物品；若要升级，用 removedItemIds 移除旧物品 + newItems 给出新版（同名但属性更强、rarity 更高）
   246→- 偷窃/赠送/典当物品：用 removedItemIds 移除
   247→- 新获物品：填入 newItems，必须含完整字段与 effects
   248→- 储物袋获得：填入 newItems（item_type=tool，effects 含 storageCapacity 加成），引擎自动扩容，玩家无需装备
   249→- 【重要】叙事中提及的物品必须落入 newItems：若 narrative 提到"父亲送我一把木剑""拾得一颗灵草"等，必须把对应物品在 newItems 中给出（含完整字段），不可只叙事不给物品
   250→
   251→【储物袋容量规则——重要】
   252→- 玩家初始无储物袋，最多只能携带 5 件物品（${invCount}/${storageCap}件${hasBag ? '，已有储物袋' : '，无储物袋'}）
   253→- 若 ${invCount} 已达或超过 ${storageCap}，本岁不可再给 newItems（除非先给一个储物袋扩容，或剧情中物品被消耗让出位置）
   254→- 储物袋本身是 tool 类物品，effects 含 storageCapacity add；获得后 capacity 增加，且储物袋不占容量
   255→- 高境界可给更高级储物袋（如"玄铁储物戒指"扩容 +30）
   256→
   257→【装备栏规则——不再限制数量上限】
   258→- 玩家「宝」页已装备栏不再是固定 5 槽位，而是数组——AI 知道玩家装备了什么就显示什么
   259→- 同类型装备可同时存在多件：玩家有十根手指，理论上可戴十个戒指；脖子上还可以挂一串；腰间可悬多件法宝
   260→- 由你（AI）判断合理性：玩家若干扰"我把一堆储物戒指用绳子串成项链戴在脖子上"，你认为合理（如玩家境界足够、有储物戒指、有绳子），就在 newEquippedItems 给出一个合成条目（如 name:"储物戒指项链", equipNote:"脖挂·储物戒指×5", effects:[storageCapacity add 50]），同时把原来的储物戒指用 removedItemIds 移除
   261→- 玩家若通过干扰想装备超过合理数量的物品（如要戴 100 个戒指），静默拒绝或减化为合理数量
   262→- 已装备物品在「宝」页用稀有度彩色显示名称，点击可看详情；卸下按钮也在该界面
   263→
   264→【奇缘异宝——特殊状态词条生成规则】
   265→玩家「宝」页除装备与储物袋外，还有「奇缘异宝」栏，展示 category=special 或 identity 的状态词条（灵宠、命格、天赋、身份、特殊体质等）。这些词条会随剧情获得或失去，AI 可通过 newStatuses 联动修改：
   266→- 灵宠/坐骑：如获灵宠，newStatuses 给出 {category:"special", name:"灵宠·小白", description:"一只通体雪白的灵狐，善感知", rarity:"rare", duration:-1, source:"山林拾得", effects:[{target_attribute:"luck",operation:"add",value:5,description:"灵宠伴身，气运微增"}]}
   267→- 命格/命途：如觉醒命格，{category:"special", name:"剑修命格", description:"天生与剑道相合", rarity:"epic", duration:-1, source:"命节点觉醒", effects:[{target_attribute:"attack",operation:"multiply",value:1.2,description:"剑器威力加成"}]}
   268→- 天赋/体质：如觉醒特殊体质，{category:"special", name:"九阳之体", description:"纯阳之体，火系功法威力倍增", rarity:"legendary", duration:-1, source:"天生", effects:[{target_attribute:"elementFire",operation:"add",value:20,description:"火属性倾向"}]}
   269→- 身份/师承：如入宗门、拜师，{category:"identity", name:"青云宗内门弟子", description:"已入青云宗内门", rarity:"uncommon", duration:-1, source:"宗门考核", effects:[]}
   270→- 临时奇遇 buff：{category:"buff", name:"灵泉淬体", description:"饮灵泉水，气血充盈", rarity:"uncommon", duration:3, source:"灵泉奇遇", effects:[{target_attribute:"maxHp",operation:"add",value:20,description:"气血上限提升"}]}
   271→注意：special/identity 类多为 duration:-1（永久）；buff 类 duration 为正数（剩余岁数）。每 3-5 岁可酌情给一个奇缘，避免过频。
   272→
   273→【修炼心得生成规则——cultivationInsight，必须每次都生成】
   274→玩家「宝」页修炼速度栏会同时展示两个内容：
   275→1. 来源条目（结构化数组，由引擎权威计算，前端按来源名称彩色高亮 + 显示具体数字）。来源包括：
   276→   - 灵根（multiply）
   277→   - 已装备功法 / 法器中的 cultivationExp 效果（multiply 或 add）
   278→   - 状态词条中的 cultivationExp 效果（multiply 或 add，如九阳之体、灵泉淬体等奇缘）
   279→   这些数字是准确的，与顶部「×倍率 +加成/岁」完全一致。你**不可**输出 cultivationFactors 字段（已从 schema 移除），
   280→   也不可在文本中编造引擎未跟踪的倍率数字——若某环境/心境真有修炼加成，应通过 newStatuses 给出 cultivationExp 效果，引擎会自动计入来源条目。
   281→2. cultivationInsight（你输出的自由文本）：60-150 字，修仙口吻，融入角色当前处境
   282→
   283→cultivationInsight 文本规则：
   284→- 60-150 字，单段纯文本（不要换行、不要 markdown、不要列表符号）
   285→- 必须在文中明确点名引擎提供的来源条目（见上方「当前修炼速度来源条目」区块）并给出准确数字
   286→  * 灵根倍率参考：无灵根 0、杂灵根 0.3、凡灵根 0.8、真灵根 1.5、天灵根 3.0、混沌灵根 5.0
   287→  * 功法倍率参考：凡品×1.2~1.5、良品×1.5~2.0、稀有×2.0~3.0、史诗×3.0~4.0、传说×4.0~5.0、神话×5.0~6.0
   288→- 末尾用一句话给出综合倍率感知，如"综合而论，修炼速度约为人之${(mult).toFixed(1)}倍"——mult 即顶部倍率（灵根×功法）
   289→- 文风：融入角色当前处境，像一位旁白在评点此子修炼之相；可描述环境氛围、心境状态，但不得编造具体倍率数字
   290→- 若本轮发生了影响修炼的事件（获功法/失法宝/觉醒体质/转境界等），心得应体现这一变化；若无所变化，可微调措辞保持鲜活
   291→- 示例（仅供文风参考，不可照抄）："土天灵根×3.0，根基已立；修习《地脉传承》×1.5，地脉之力加持。腰悬聚灵佩+5，灵气汇聚。近日道心通明，唯年尚幼，气血未丰。综合而论，修炼速度约人之4.5倍。"
   292→- 无灵根示例："凡人之躯，难引天地灵气。虽勤练不辍，然经脉滞涩，修为寸进。综合而论，修炼速度约人之零倍。"
   293→- 严禁出现 JSON 转义问题：文本内不得出现裸双引号、裸换行符；书名号《》可以用；若需引用功法名，直接写书名即可`;
   294→}

   411→async function callLLM(systemPrompt: string, userPrompt: string, scenePrompt: string): Promise<any> {
   412→  const zai = await getZAI();
   413→  const fullSystem = `${systemPrompt}\n\n${scenePrompt}`;
   414→  try {
   415→    const completion = await zai.chat.completions.create({
   416→      messages: [
   417→        { role: 'system', content: fullSystem },
   418→        { role: 'user', content: userPrompt },
   419→      ],
   420→      thinking: { type: 'disabled' },
   421→    });
   422→    const content = completion.choices[0]?.message?.content || '';
   423→    return parseJSON(content);
   424→  } catch (err: any) {
   425→    console.error('LLM call failed:', err?.message || err);
   426→    throw err;
   427→  }
   428→}

   430→// 从 LLM 输出中提取 JSON（兼容 ```json ``` 包装、未转义字符、尾随逗号等常见问题）
   431→function parseJSON(content: string): any {
   432→  let s = content.trim();
   433→  // 移除 markdown 代码块
   434→  const fenceMatch = s.match(/```(?:json)?\s*([\s\S]*?)```/);
   435→  if (fenceMatch) {
   436→    s = fenceMatch[1].trim();
   437→  }
   438→  // 找第一个 { 到最后一个 }
   439→  const start = s.indexOf('{');
   440→  const end = s.lastIndexOf('}');
   441→  if (start >= 0 && end > start) {
   442→    s = s.slice(start, end + 1);
   443→  }
   444→  // 尝试直接解析
   445→  try {
   446→    return JSON.parse(s);
   447→  } catch {
   448→    // 失败则尝试修复常见问题
   449→    return JSON.parse(repairJSON(s));
   450→  }
   451→}
   452→
   453→// 修复 LLM 输出 JSON 的常见问题：
   454→// 1. 字符串值内未转义的双引号（如 narrative: "他说"你好"了"）
   455→// 2. 字符串值内的裸换行符（JSON 标准要求 \n）
   456→// 3. 尾随逗号
   457→// 策略：逐字符扫描，仅在字符串外应用结构修复，字符串内转义裸引号/换行
   458→function repairJSON(s: string): string {
   459→  let out = '';
   460→  let inStr = false;
   461→  let escape = false;
   462→  for (let i = 0; i < s.length; i++) {
   463→    const ch = s[i];
   464→    if (inStr) {
   465→      if (escape) {
   466→        out += ch;
   467→        escape = false;
   468→        continue;
   469→      }
   470→      if (ch === '\\') {
   471→        out += ch;
   472→        escape = true;
   473→        continue;
   474→      }
   475→      if (ch === '"') {
   476→        // 判断这是字符串结束还是裸引号：
   477→        // 看后面（跳过空格）是否是 , } ] : —— 若是则视为字符串结束，否则视为裸引号需转义
   478→        let j = i + 1;
   479→        while (j < s.length && (s[j] === ' ' || s[j] === '\t')) j++;
   480→        const nextCh = s[j];
   481→        if (nextCh === ',' || nextCh === '}' || nextCh === ']' || nextCh === ':' || nextCh === undefined) {
   482→          out += ch;
   483→          inStr = false;
   484→        } else {
   485→          // 裸引号，转义
   486→          out += '\\"';
   487→        }
   488→        continue;
   489→      }
   490→      if (ch === '\n') {
   491→        out += '\\n';
   492→        continue;
   493→      }
   494→      if (ch === '\r') {
   495→        out += '\\r';
   496→        continue;
   497→      }
   498→      if (ch === '\t') {
   499→        out += '\\t';
   500→        continue;
   501→      }
   502→      out += ch;
   503→    } else {
   504→      if (ch === '"') {
   505→        inStr = true;
   506→        out += ch;
   507→      } else {
   508→        out += ch;
   509→      }
   510→    }
   511→  }
   512→  // 移除尾随逗号（,} 或 ,]）
   513→  out = out.replace(/,(\s*[}\]])/g, '$1');
   514→  return out;
   515→}

   519→// 后处理：修正 narrative 中主角年龄数字（AI 偶发幻觉会把年龄写错，如3岁写成"四岁"）
   520→// 策略：匹配紧邻主角名或句首的"数字+岁"，若数字≠正确年龄则替换为正确年龄（支持中文数字与阿拉伯数字）
   521→const ZH_DIGIT: Record<string, number> = { 零:0,一:1,二:2,两:2,三:3,四:4,五:5,六:6,七:7,八:8,九:9,十:10 };
   522→const ZH_AGE: Record<number, string> = { 0:'零',1:'一',2:'二',3:'三',4:'四',5:'五',6:'六',7:'七',8:'八',9:'九',10:'十' };

   524→// 中文数字（0-99）转 number，非中文数字返回 null
   525→function zhAgeToNum(s: string): number | null {
   526→  if (/^\d+$/.test(s)) return parseInt(s, 10);
   527→  if (!/^[\u96f6\u4e00\u4e8c\u4e24\u4e09\u56db\u4e94\u516d\u4e03\u516b\u4e5d\u5341]+$/.test(s)) return null;
   528→  if (s === '十') return 10;
   529→  if (s.startsWith('十')) return 10 + (ZH_DIGIT[s[1]] ?? 0);
   530→  if (s.endsWith('十')) return (ZH_DIGIT[s[0]] ?? 0) * 10;
   531→  if (s.includes('十')) {
   532→    const parts = s.split('十');
   533→    return (ZH_DIGIT[parts[0]] ?? 0) * 10 + (ZH_DIGIT[parts[1]] ?? 0);
   534→  }
   535→  // 单字
   536→  if (s.length === 1 && s in ZH_DIGIT) return ZH_DIGIT[s];
   537→  return null;
   538→}
   539→
   540→// number → 中文数字（0-99），用于替换文案
   541→function numToZhAge(n: number): string {
   542→  if (n in ZH_AGE) return ZH_AGE[n];
   543→  if (n < 20) return '十' + (ZH_AGE[n - 10] ?? String(n - 10));
   544→  const tens = Math.floor(n / 10);
   545→  const ones = n % 10;
   546→  return (ZH_AGE[tens] ?? String(tens)) + '十' + (ones ? (ZH_AGE[ones] ?? String(ones)) : '');
   547→}

   549→function fixNarrativeAge(narrative: string, correctAge: number, charName: string): string {
   550→  if (!narrative) return narrative;
   551→  let out = narrative;
   552→  // 匹配"数字+岁+的+角色名"或"数字+岁+的+主角"等明确指代主角的模式
   553→  // 中文数字或阿拉伯数字，0-99
   554→  const numPat = '([0-9]+|[零一二三四五六七八九十]{1,3})';
   555→  // 模式1："X岁的{name}" / "X岁那年{name}" / "X岁时，{name}"
   556→  const re1 = new RegExp(`${numPat}岁(?:的|那年|时[，,]?)(.{0,2}?)${escapeRegExp(charName)}`, 'g');
   557→  // 模式2：句首"X岁，"或"X岁的"开头（通常指代主角）
   558→  const re2 = new RegExp(`^${numPat}岁(?:的|，|时|那年)`, 'g');
   559→  // 模式3："{name}...X岁" 紧邻（中间部分用非捕获组，保证数字是第一个捕获组）
   560→  const re3 = new RegExp(`${escapeRegExp(charName)}(?:[\\s\\S]{0,8}?)${numPat}岁`, 'g');
   561→
   562→  const replaceNum = (m: string, g1: string, ...rest: any[]) => {
   563→    const num = zhAgeToNum(g1);
   564→    if (num === null || num === correctAge) return m; // 无法解析或本来就对，不动
   565→    if (num < 0 || num > 150) return m; // 明显是别的语境（如寿元、年份）
   566→    // 替换为正确年龄（保持原格式：若原文是阿拉伯数字就用阿拉伯，否则用中文）
   567→    const isArabic = /[0-9]/.test(g1);
   568→    const replacement = isArabic ? String(correctAge) : numToZhAge(correctAge);
   569→    return m.replace(g1, replacement);
   570→  };
   571→
   572→  out = out.replace(re1, replaceNum);
   573→  out = out.replace(re2, replaceNum);
   574→  out = out.replace(re3, replaceNum);
   575→  return out;
   576→}
   577→
   578→function escapeRegExp(s: string): string {
   579→  return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
   580→}

   845→// 引擎权威：灵根类型与五行组合由后端按概率随机生成（LLM 不可自由发挥，避免每次重生结果趋同）
   846→// 灵根类型按 rarity 权重抽取（none30/mixed25/common20/pure15/heavenly8/chaos2）
   847→
   848→const ALL_ELEMENTS: Element[] = ['metal', 'wood', 'water', 'fire', 'earth'];
   849→
   850→function rollSpiritualRoot(): SpiritualRoot {
   851→  const entries = Object.entries(SPIRITUAL_ROOTS) as [SpiritualRoot, { rarity: number }][];
   852→  const total = entries.reduce((s, [, v]) => s + v.rarity, 0);
   853→  let r = Math.random() * total;
   854→  for (const [k, v] of entries) {
   855→    r -= v.rarity;
   856→    if (r <= 0) return k;
   857→  }
   858→  return 'mixed';
   859→}
   860→
   861→// 根据灵根类型随机生成五行倾向（哪些元素突出），并返回初始五行数值
   862→// 返回 { elements, picked }：picked 为本次突出的元素列表（用于让 LLM 生成 rootDetail）
   863→function rollElements(root: SpiritualRoot): { elements: Record<Element, number>; picked: Element[] } {
   864→  const shuffle = <T,>(arr: T[]): T[] => {
   865→    const a = [...arr];
   866→    for (let i = a.length - 1; i > 0; i--) {
   867→      const j = Math.floor(Math.random() * (i + 1));
   868→      [a[i], a[j]] = [a[j], a[i]];
   869→    }
   870→    return a;
   871→  };
   872→
   873→  switch (root) {
   874→    case 'none':
   875→      // 无灵根：五行都很低
   876→      return { elements: { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 }, picked: [] };
   877→    case 'mixed':
   878→      // 杂灵根：五行皆中等
   879→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   880→    case 'common': {
   881→      // 凡灵根：2-3 种属性突出
   882→      const count = 2 + Math.floor(Math.random() * 2); // 2 或 3
   883→      const picked = shuffle(ALL_ELEMENTS).slice(0, count);
   884→      const elements: Record<Element, number> = { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 };
   885→      for (const e of picked) elements[e] = 30 + Math.floor(Math.random() * 11); // 30-40
   886→      return { elements, picked };
   887→    }
   888→    case 'pure': {
   889→      // 真灵根：单属性突出
   890→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   891→      const elements: Record<Element, number> = { metal: 5, wood: 5, water: 5, fire: 5, earth: 5 };
   892→      elements[picked[0]] = 50 + Math.floor(Math.random() * 11); // 50-60
   893→      return { elements, picked };
   894→    }
   895→    case 'heavenly': {
   896→      // 天灵根：单属性极突出
   897→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   898→      const elements: Record<Element, number> = { metal: 3, wood: 3, water: 3, fire: 3, earth: 3 };
   899→      elements[picked[0]] = 70 + Math.floor(Math.random() * 11); // 70-80
   900→      return { elements, picked };
   901→    }
   902→    case 'chaos':
   903→      // 混沌灵根：五行皆高
   904→      return { elements: { metal: 45, wood: 45, water: 45, fire: 45, earth: 45 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   905→    default:
   906→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: [] };
   907→  }
   908→}
   909→
   910→// 把元素列表转中文描述（如 ["fire","wood"] → "火木"）
   911→function elementsToZh(els: Element[]): string {
   912→  return els.map(e => ELEMENTS[e].name).join('');
   913→}
