   115→function buildAdvancePrompt(ctx: EngineStateContext, isFateNode: boolean): string {
   116→  const sc = ctx.character;
   117→  const elements = `金${sc.elements.metal}/木${sc.elements.wood}/水${sc.elements.water}/火${sc.elements.fire}/土${sc.elements.earth}`;
   118→  const statusList = ctx.activeStatuses.length
   119→    ? ctx.activeStatuses.map(s => `- ${s.name}（${s.category}，${s.rarity}）：${s.description}`).join('\n')
   120→    : '无';
   121→  const invList = ctx.inventory.length
   122→    ? ctx.inventory.map(i => `- [id:${i.id}] ${i.name}（${i.rarity}/${i.item_type}）：${i.description}${i.effects?.length ? '；效果：' + i.effects.map(e => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   123→    : '无';
   124→  const eqList = ctx.equipped && Object.keys(ctx.equipped).length
   125→    ? Object.entries(ctx.equipped).map(([slot, it]: [string, any]) => `- [槽位:${slot}][id:${it.id}] ${it.name}（${it.rarity}/${it.item_type}）：${it.description}${it.effects?.length ? '；效果：' + it.effects.map((e: any) => `${e.operation === 'add' ? '+' : '×'}${e.value} ${e.target_attribute}`).join('，') : ''}`).join('\n')
   126→    : '无';
   127→  const recentEvts = ctx.recentEvents.length
   128→    ? ctx.recentEvents.map(e => `${e.age}岁：${e.title}——${e.narrative.slice(0, 80)}`).join('\n')
   129→    : '无';
   130→  const memory = ctx.longTermMemory.length
   131→    ? ctx.longTermMemory.map(m => `- ${m}`).join('\n')
   132→    : '无';
   133→  const mult = ctx.cultivationMultiplier || 0;
   134→  const multDesc = mult > 0 ? `${mult.toFixed(2)}倍（已含灵根与功法加成）` : '0（无灵根，无法修炼）';
   135→  const curInsight = ctx.cultivationInsight || '';
   136→
   137→  return `【状态快照区】
   138→角色：${sc.name}（${sc.gender === 'male' ? '男' : '女'}），${sc.age}岁
   139→寿元：${sc.lifespan}岁（剩余约${sc.lifespan - sc.age}岁）
   140→灵根：${sc.rootDetail || sc.spiritualRoot}
   141→境界：${sc.realmName}${sc.realmLevel > 0 ? `（${sc.realmLevel}层）` : ''}
   142→修为：${sc.cultivationExp}/${sc.expToBreak}（修炼速度：${multDesc}，你给出的 cultivationExp 正向增量会被该倍率放大）
   143→五行倾向：${elements}
   144→生命：${sc.hp}/${sc.maxHp}  灵力：${sc.mp}/${sc.maxMp}
   145→攻击：${sc.attack}  防御：${sc.defense}  速度：${sc.speed}
   146→气运：${sc.luck}  悟性：${sc.comprehension}
   147→灵石：${sc.spiritStones}  声望：${sc.reputation}
   148→宗门：${sc.faction || '散修'}  师承：${sc.master || '无'}  所在：${sc.location}
   149→当前状态词条：
   150→${statusList}
   151→背包（储物袋，物品 id 已标注）：
   152→${invList}
   153→已装备（槽位/物品 id 已标注）：
   154→${eqList}
   155→
   156→【当前修炼心得】（玩家「宝」页修炼速度栏展示文本，由你上一轮生成，本轮可更新）
   157→${curInsight || '（尚未生成，本轮请首次生成）'}
   158→
   159→【记忆检索区】长期记忆：
   160→${memory}
   161→
   162→【短期对话区】最近事件：
   163→${recentEvts}
   164→
   165→${isFateNode && ctx.nextFateNode ? `【命节点触发】本轮应触发命节点 #${ctx.nextFateNode.index}「${ctx.nextFateNode.name}」！
   166→此命节点对应境界：${ctx.nextFateNode.realm}
   167→本命节点必须 hasChoice=true，生成 3-4 个有意义的选项供玩家选择。
   168→narrative 应铺垫此命节点的核心冲突，choice.prompt 描述抉择情境。
   169→此命节点完成后角色将永久性改变（获得身份、师承、命格等）。` : '【命节点】本轮无命节点触发，生成普通年龄事件。'}
   170→
   171→请生成 JSON，schema 如下：
   172→{
   173→  "title": "事件标题（≤16字）",
   174→  "narrative": "叙事正文（100-300字，命节点200-400字）",
   175→  "eventType": "normal | fate_node | choice | combat | breakthrough | death | ascension",
   176→  "changes": [{"attribute":"cultivationExp","delta":10,"reason":"修炼精进"}],
   177→  "newStatuses": [],
   178→  "newItems": [],
   179→  "removedItemIds": [],
   180→  "memory": "本岁关键事件一句话摘要，写入长期记忆",
   181→  "cultivationInsight": "修炼心得文本（60-150字，见下方生成规则）",
   182→  "hasChoice": false,
   183→  "choice": null,
   184→  "triggeredBreakthrough": false,
   185→  "causedDeath": false,
   186→  "causedAscension": false
   187→}
   188→
   189→可修改属性白名单：${ctx.availableAttributes.join(', ')}
   190→注意：attribute 必须在白名单内；delta 合理（普通事件 -50~+100，奇遇 -200~+500）；newStatuses 与 newItems 给出完整字段。
   191→严禁修改 age（年龄由天道推进，每岁固定 +1，AI 不得在 changes 中包含 age）。
   192→removedItemIds：若事件中某物品被破坏/消耗/丢失（如战斗中兵器损毁、丹药被服用、法宝碎裂），把该物品的 id 填入此数组，引擎会自动从储物袋或已装备中移除并反向结算属性。无则留空数组。
   193→
   194→statusEntry 结构：{id,name,description,category(attribute/skill/buff/debuff/special/identity/quest/environment),rarity(common/uncommon/rare/epic/legendary/mythic),duration(-1永久/正数为剩余岁数),source,effects:[{target_attribute,operation(add/multiply/override/cap/floor/trigger),value,description}]}
   195→
   196→itemEntry 结构：{id,name,description,item_type,rarity,effects:[...],source}
   197→【物品生成规则——必须严格遵守】
   198→- item_type 取值：weapon(兵器)/armor(防具)/accessory(饰物)/artifact(法宝)/consumable(丹药)/material(材料)/tool(器具)/scripture(功法)
   199→- rarity 必须与玩家境界匹配，严禁越级给高级物品：
   200→  * 凡人/炼气期：common 或 uncommon
   201→  * 筑基期：uncommon 或 rare
   202→  * 金丹期：rare 或 epic
   203→  * 元婴期：epic 或 legendary
   204→  * 化神及以上：legendary 或 mythic
   205→- effects 必须给出且符合物品类型语义：
   206→  * weapon(兵器)：effects 用 add operation，target_attribute 为 attack（如 +10 attack）；高阶兵器可加 speed 或 hp
   207→  * armor(防具)：effects 用 add，target 为 defense（如 +8 defense）；高阶可加 maxHp
   208→  * accessory(饰物)：effects 用 add，target 为 luck/comprehension/maxMp 等
   209→  * artifact(法宝)：effects 用 add 或 multiply，target 为 attack/defense/speed/cultivationExp 等；高阶法宝可有 multiply 效果
   210→  * consumable(丹药)：effects 用 add，target 为 hp/mp/cultivationExp/lifespan 等；服用后消失
   211→  * scripture(功法)：effects 必含一条 multiply cultivationExp（修炼倍率，凡品×1.2~1.5、良品×1.5~2.0、稀有×2.0~3.0、史诗×3.0~4.0、传说×4.0~5.0、神话×5.0~6.0）；可附带 add cultivationExp 等被动
   212→  * material/tool：通常无 effects，仅作剧情道具
   213→- id 格式：item_<类型缩写>_<4位随机>，如 item_wpn_a3f2、item_scr_b8c1、item_pil_d2e4。同一事件多个物品 id 不可重复。
   214→- name：符合修仙世界风格（如"青锋剑""玄铁甲""聚气丹""引气诀""紫金葫芦"），≤8字
   215→- description：10-40字，描述外观/功效/来历
   216→- source：必须填写，简述获得方式（如"宗门发放""秘境拾得""炼丹炉出""战胜妖兽所得"）
   217→
   218→【物品修改规则——AI 联动】
   219→- 战斗中兵器/防具/法宝可能损毁：把对应物品 id 填入 removedItemIds
   220→- 丹药服用消耗：填入 removedItemIds
   221→- 物品升级/精炼：不要在 newItems 重复给已有物品；若要升级，用 removedItemIds 移除旧物品 + newItems 给出新版（同名但属性更强、rarity 更高）
   222→- 偷窃/赠送/典当物品：用 removedItemIds 移除
   223→- 新获物品：填入 newItems，必须含完整字段与 effects
   224→
   225→【奇缘异宝——特殊状态词条生成规则】
   226→玩家「宝」页除装备与储物袋外，还有「奇缘异宝」栏，展示 category=special 或 identity 的状态词条（灵宠、命格、天赋、身份、特殊体质等）。这些词条会随剧情获得或失去，AI 可通过 newStatuses 联动修改：
   227→- 灵宠/坐骑：如获灵宠，newStatuses 给出 {category:"special", name:"灵宠·小白", description:"一只通体雪白的灵狐，善感知", rarity:"rare", duration:-1, source:"山林拾得", effects:[{target_attribute:"luck",operation:"add",value:5,description:"灵宠伴身，气运微增"}]}
   228→- 命格/命途：如觉醒命格，{category:"special", name:"剑修命格", description:"天生与剑道相合", rarity:"epic", duration:-1, source:"命节点觉醒", effects:[{target_attribute:"attack",operation:"multiply",value:1.2,description:"剑器威力加成"}]}
   229→- 天赋/体质：如觉醒特殊体质，{category:"special", name:"九阳之体", description:"纯阳之体，火系功法威力倍增", rarity:"legendary", duration:-1, source:"天生", effects:[{target_attribute:"elementFire",operation:"add",value:20,description:"火属性倾向"}]}
   230→- 身份/师承：如入宗门、拜师，{category:"identity", name:"青云宗内门弟子", description:"已入青云宗内门", rarity:"uncommon", duration:-1, source:"宗门考核", effects:[]}
   231→- 临时奇遇 buff：{category:"buff", name:"灵泉淬体", description:"饮灵泉水，气血充盈", rarity:"uncommon", duration:3, source:"灵泉奇遇", effects:[{target_attribute:"maxHp",operation:"add",value:20,description:"气血上限提升"}]}
   232→注意：special/identity 类多为 duration:-1（永久）；buff 类 duration 为正数（剩余岁数）。每 3-5 岁可酌情给一个奇缘，避免过频。
   233→
   234→【修炼心得生成规则——cultivationInsight 字段，必须每次都生成】
   235→玩家「宝」页的修炼速度栏不再显示固定公式，而是展示你生成的这段文本。请根据当前角色全状态，用修仙世界口吻写一段描述当前影响修炼速度的因素。显示规则：
   236→- 长度 60-150 字，单段纯文本（不要换行、不要 markdown、不要列表符号）

   335→async function callLLM(systemPrompt: string, userPrompt: string, scenePrompt: string): Promise<any> {
   336→  const zai = await getZAI();
   337→  const fullSystem = `${systemPrompt}\n\n${scenePrompt}`;
   338→  try {
   339→    const completion = await zai.chat.completions.create({
   340→      messages: [
   341→        { role: 'system', content: fullSystem },
   342→        { role: 'user', content: userPrompt },
   343→      ],
   344→      thinking: { type: 'disabled' },
   345→    });
   346→    const content = completion.choices[0]?.message?.content || '';
   347→    return parseJSON(content);
   348→  } catch (err: any) {
   349→    console.error('LLM call failed:', err?.message || err);
   350→    throw err;
   351→  }
   352→}

   354→// 从 LLM 输出中提取 JSON（兼容 ```json ``` 包装、未转义字符、尾随逗号等常见问题）
   355→function parseJSON(content: string): any {
   356→  let s = content.trim();
   357→  // 移除 markdown 代码块
   358→  const fenceMatch = s.match(/```(?:json)?\s*([\s\S]*?)```/);
   359→  if (fenceMatch) {
   360→    s = fenceMatch[1].trim();
   361→  }
   362→  // 找第一个 { 到最后一个 }
   363→  const start = s.indexOf('{');
   364→  const end = s.lastIndexOf('}');
   365→  if (start >= 0 && end > start) {
   366→    s = s.slice(start, end + 1);
   367→  }
   368→  // 尝试直接解析
   369→  try {
   370→    return JSON.parse(s);
   371→  } catch {
   372→    // 失败则尝试修复常见问题
   373→    return JSON.parse(repairJSON(s));
   374→  }
   375→}
   376→
   377→// 修复 LLM 输出 JSON 的常见问题：
   378→// 1. 字符串值内未转义的双引号（如 narrative: "他说"你好"了"）
   379→// 2. 字符串值内的裸换行符（JSON 标准要求 \n）
   380→// 3. 尾随逗号
   381→// 策略：逐字符扫描，仅在字符串外应用结构修复，字符串内转义裸引号/换行
   382→function repairJSON(s: string): string {
   383→  let out = '';
   384→  let inStr = false;
   385→  let escape = false;
   386→  for (let i = 0; i < s.length; i++) {
   387→    const ch = s[i];
   388→    if (inStr) {
   389→      if (escape) {
   390→        out += ch;
   391→        escape = false;
   392→        continue;
   393→      }
   394→      if (ch === '\\') {
   395→        out += ch;
   396→        escape = true;
   397→        continue;
   398→      }
   399→      if (ch === '"') {
   400→        // 判断这是字符串结束还是裸引号：
   401→        // 看后面（跳过空格）是否是 , } ] : —— 若是则视为字符串结束，否则视为裸引号需转义
   402→        let j = i + 1;
   403→        while (j < s.length && (s[j] === ' ' || s[j] === '\t')) j++;
   404→        const nextCh = s[j];
   405→        if (nextCh === ',' || nextCh === '}' || nextCh === ']' || nextCh === ':' || nextCh === undefined) {
   406→          out += ch;
   407→          inStr = false;
   408→        } else {
   409→          // 裸引号，转义
   410→          out += '\\"';
   411→        }
   412→        continue;
   413→      }
   414→      if (ch === '\n') {
   415→        out += '\\n';
   416→        continue;
   417→      }
   418→      if (ch === '\r') {
   419→        out += '\\r';
   420→        continue;
   421→      }
   422→      if (ch === '\t') {
   423→        out += '\\t';
   424→        continue;
   425→      }
   426→      out += ch;
   427→    } else {
   428→      if (ch === '"') {
   429→        inStr = true;
   430→        out += ch;
   431→      } else {
   432→        out += ch;
   433→      }
   434→    }
   435→  }
   436→  // 移除尾随逗号（,} 或 ,]）
   437→  out = out.replace(/,(\s*[}\]])/g, '$1');
   438→  return out;
   439→}

   443→// 后处理：修正 narrative 中主角年龄数字（AI 偶发幻觉会把年龄写错，如3岁写成"四岁"）
   444→// 策略：匹配紧邻主角名或句首的"数字+岁"，若数字≠正确年龄则替换为正确年龄（支持中文数字与阿拉伯数字）
   445→const ZH_DIGIT: Record<string, number> = { 零:0,一:1,二:2,两:2,三:3,四:4,五:5,六:6,七:7,八:8,九:9,十:10 };
   446→const ZH_AGE: Record<number, string> = { 0:'零',1:'一',2:'二',3:'三',4:'四',5:'五',6:'六',7:'七',8:'八',9:'九',10:'十' };
   447→
   448→// 中文数字（0-99）转 number，非中文数字返回 null
   449→function zhAgeToNum(s: string): number | null {
   450→  if (/^\d+$/.test(s)) return parseInt(s, 10);
   451→  if (!/^[\u96f6\u4e00\u4e8c\u4e24\u4e09\u56db\u4e94\u516d\u4e03\u516b\u4e5d\u5341]+$/.test(s)) return null;
   452→  if (s === '十') return 10;
   453→  if (s.startsWith('十')) return 10 + (ZH_DIGIT[s[1]] ?? 0);
   454→  if (s.endsWith('十')) return (ZH_DIGIT[s[0]] ?? 0) * 10;
   455→  if (s.includes('十')) {
   456→    const parts = s.split('十');
   457→    return (ZH_DIGIT[parts[0]] ?? 0) * 10 + (ZH_DIGIT[parts[1]] ?? 0);
   458→  }
   459→  // 单字
   460→  if (s.length === 1 && s in ZH_DIGIT) return ZH_DIGIT[s];
   461→  return null;
   462→}
   463→
   464→// number → 中文数字（0-99），用于替换文案
   465→function numToZhAge(n: number): string {
   466→  if (n in ZH_AGE) return ZH_AGE[n];
   467→  if (n < 20) return '十' + (ZH_AGE[n - 10] ?? String(n - 10));
   468→  const tens = Math.floor(n / 10);
   469→  const ones = n % 10;
   470→  return (ZH_AGE[tens] ?? String(tens)) + '十' + (ones ? (ZH_AGE[ones] ?? String(ones)) : '');
   471→}

   473→function fixNarrativeAge(narrative: string, correctAge: number, charName: string): string {
   474→  if (!narrative) return narrative;
   475→  let out = narrative;
   476→  // 匹配"数字+岁+的+角色名"或"数字+岁+的+主角"等明确指代主角的模式
   477→  // 中文数字或阿拉伯数字，0-99
   478→  const numPat = '([0-9]+|[零一二三四五六七八九十]{1,3})';
   479→  // 模式1："X岁的{name}" / "X岁那年{name}" / "X岁时，{name}"
   480→  const re1 = new RegExp(`${numPat}岁(?:的|那年|时[，,]?)(.{0,2}?)${escapeRegExp(charName)}`, 'g');
   481→  // 模式2：句首"X岁，"或"X岁的"开头（通常指代主角）
   482→  const re2 = new RegExp(`^${numPat}岁(?:的|，|时|那年)`, 'g');
   483→  // 模式3："{name}...X岁" 紧邻（中间部分用非捕获组，保证数字是第一个捕获组）
   484→  const re3 = new RegExp(`${escapeRegExp(charName)}(?:[\\s\\S]{0,8}?)${numPat}岁`, 'g');
   485→
   486→  const replaceNum = (m: string, g1: string, ...rest: any[]) => {
   487→    const num = zhAgeToNum(g1);
   488→    if (num === null || num === correctAge) return m; // 无法解析或本来就对，不动
   489→    if (num < 0 || num > 150) return m; // 明显是别的语境（如寿元、年份）
   490→    // 替换为正确年龄（保持原格式：若原文是阿拉伯数字就用阿拉伯，否则用中文）
   491→    const isArabic = /[0-9]/.test(g1);
   492→    const replacement = isArabic ? String(correctAge) : numToZhAge(correctAge);
   493→    return m.replace(g1, replacement);
   494→  };
   495→
   496→  out = out.replace(re1, replaceNum);
   497→  out = out.replace(re2, replaceNum);
   498→  out = out.replace(re3, replaceNum);
   499→  return out;
   500→}
   501→
   502→function escapeRegExp(s: string): string {
   503→  return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
   504→}

   615→        attribute: String(c.attribute || ''),
   616→        delta: Number(c.delta) || 0,
   617→        reason: String(c.reason || ''),
   618→      })).filter((c: AttributeChange) => c.attribute)
   619→    : [];
   620→
   621→  const { statuses, items } = ensureUniqueIds(
   622→    accepted && Array.isArray(raw?.newStatuses) ? raw.newStatuses : [],
   623→    accepted && Array.isArray(raw?.newItems) ? raw.newItems : []
   624→  );
   625→
   626→  return {
   627→    classification: cls,
   628→    accepted,
   629→    narrative: String(raw?.narrative || (accepted ? '天道如是。' : '世界自按其轨运行。')),
   630→    changes,
   631→    newStatuses: statuses,
   632→    newItems: items,
   633→    removedItemIds: accepted && Array.isArray(raw?.removedItemIds) ? raw.removedItemIds.map((x: any) => String(x)).filter(Boolean) : [],
   634→    memory: accepted ? String(raw?.memory || '') : '',
   635→    cultivationInsight: accepted && raw?.cultivationInsight ? String(raw.cultivationInsight).slice(0, 400) : '',
   636→    ageAdvance: accepted ? Math.max(0, Math.min(5, Number(raw?.ageAdvance) || 0)) : 0,
   637→  };
   638→}

   640→// ==================== 出生事件生成 ====================
   641→
   642→// 引擎权威：灵根类型与五行组合由后端按概率随机生成（LLM 不可自由发挥，避免每次重生结果趋同）
   643→// 灵根类型按 rarity 权重抽取（none30/mixed25/common20/pure15/heavenly8/chaos2）
   644→
   645→const ALL_ELEMENTS: Element[] = ['metal', 'wood', 'water', 'fire', 'earth'];
   646→
   647→function rollSpiritualRoot(): SpiritualRoot {
   648→  const entries = Object.entries(SPIRITUAL_ROOTS) as [SpiritualRoot, { rarity: number }][];
   649→  const total = entries.reduce((s, [, v]) => s + v.rarity, 0);
   650→  let r = Math.random() * total;
   651→  for (const [k, v] of entries) {
   652→    r -= v.rarity;
   653→    if (r <= 0) return k;
   654→  }
   655→  return 'mixed';
   656→}
   657→
   658→// 根据灵根类型随机生成五行倾向（哪些元素突出），并返回初始五行数值
   659→// 返回 { elements, picked }：picked 为本次突出的元素列表（用于让 LLM 生成 rootDetail）
   660→function rollElements(root: SpiritualRoot): { elements: Record<Element, number>; picked: Element[] } {
   661→  const shuffle = <T,>(arr: T[]): T[] => {
   662→    const a = [...arr];
   663→    for (let i = a.length - 1; i > 0; i--) {
   664→      const j = Math.floor(Math.random() * (i + 1));
   665→      [a[i], a[j]] = [a[j], a[i]];
   666→    }
   667→    return a;
   668→  };
   669→
   670→  switch (root) {
   671→    case 'none':
   672→      // 无灵根：五行都很低
   673→      return { elements: { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 }, picked: [] };
   674→    case 'mixed':
   675→      // 杂灵根：五行皆中等
   676→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   677→    case 'common': {
   678→      // 凡灵根：2-3 种属性突出
   679→      const count = 2 + Math.floor(Math.random() * 2); // 2 或 3
   680→      const picked = shuffle(ALL_ELEMENTS).slice(0, count);
   681→      const elements: Record<Element, number> = { metal: 8, wood: 8, water: 8, fire: 8, earth: 8 };
   682→      for (const e of picked) elements[e] = 30 + Math.floor(Math.random() * 11); // 30-40
   683→      return { elements, picked };
   684→    }
   685→    case 'pure': {
   686→      // 真灵根：单属性突出
   687→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   688→      const elements: Record<Element, number> = { metal: 5, wood: 5, water: 5, fire: 5, earth: 5 };
   689→      elements[picked[0]] = 50 + Math.floor(Math.random() * 11); // 50-60
   690→      return { elements, picked };
   691→    }
   692→    case 'heavenly': {
   693→      // 天灵根：单属性极突出
   694→      const picked = shuffle(ALL_ELEMENTS).slice(0, 1);
   695→      const elements: Record<Element, number> = { metal: 3, wood: 3, water: 3, fire: 3, earth: 3 };
   696→      elements[picked[0]] = 70 + Math.floor(Math.random() * 11); // 70-80
   697→      return { elements, picked };
   698→    }
   699→    case 'chaos':
   700→      // 混沌灵根：五行皆高
   701→      return { elements: { metal: 45, wood: 45, water: 45, fire: 45, earth: 45 }, picked: shuffle(ALL_ELEMENTS).slice(0, 5) };
   702→    default:
   703→      return { elements: { metal: 18, wood: 18, water: 18, fire: 18, earth: 18 }, picked: [] };
   704→  }
   705→}
   706→
   707→// 把元素列表转中文描述（如 ["fire","wood"] → "火木"）
   708→function elementsToZh(els: Element[]): string {
   709→  return els.map(e => ELEMENTS[e].name).join('');
   710→}

   720→  // 后端 roll 出的五行数值（route 层用来覆盖默认 20/20/20/20/20）
